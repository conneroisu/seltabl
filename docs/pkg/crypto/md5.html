		Package md5 implements the MD5 hash algorithm as defined in RFC 1321.
		
		MD5 is cryptographically broken and should not be used for secure
		applications.

			if err != nil {
				log.Fatal(err)
			}
			defer f.Close()
		
			h := md5.New()
			if _, err := io.Copy(h, f); err != nil {
				log.Fatal(err)
			}
		
			fmt.Printf("%x", h.Sum(nil))
		}
